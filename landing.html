Quick Start

from hyperbolica import Complex, hyperbolic_distance, HyperbolicTerrain, HyperbolicTiling

# Distance in the Poincare disk
a = Complex(0.3, 0.2)
b = Complex(-0.5, 0.1)
print(hyperbolic_distance(a, b))  # 1.832...

# Generate infinite terrain
terrain = HyperbolicTerrain(seed=42, palette="volcanic")
height = terrain.height_at(Complex(0.5, 0.3))
grid = terrain.height_map(resolution=256)

# Escher-style tessellation
tiling = HyperbolicTiling(p=7, q=3)
svg = tiling.to_svg(800)  # publication-quality SVG
print(f"Generated {tiling.num_tiles} tiles")
